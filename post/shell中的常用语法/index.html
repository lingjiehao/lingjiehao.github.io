获取文件名/文件扩展名

~% FILE="example.tar.gz"
~% echo "${FILE%%.*}"
example
~% echo "${FILE%.*}"
example.tar
~% echo "${FILE#*.}"
tar.gz
~% echo "${FILE##*.}"
gz

创建临时文件夹

# 创建临时文件夹
$ mktemp -d
/tmp/tmp.vwjuFmclTR

# 创建临时文件夹，并指定模板
$ mktemp -d -t test-XXXXX
/tmp/test-4kceq
