VERSION_CODENAME=bullseye
MIRROR_URL=mirrors.ustc.edu.cn
HTTP_OR_HTTPS=https

cat <<EOF | sudo tee /etc/apt/sources.list
deb ${HTTP_OR_HTTPS}://${MIRROR_URL}/ubuntu/ ${VERSION_CODENAME} main restricted universe multiverse
deb ${HTTP_OR_HTTPS}://${MIRROR_URL}/ubuntu/ ${VERSION_CODENAME}-updates main restricted universe multiverse
deb ${HTTP_OR_HTTPS}://${MIRROR_URL}/ubuntu/ ${VERSION_CODENAME}-backports main restricted universe multiverse
deb ${HTTP_OR_HTTPS}://${MIRROR_URL}/ubuntu/ ${VERSION_CODENAME}-security main restricted universe multiverse

EOF
